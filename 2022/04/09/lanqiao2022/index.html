对于 20% 的数据，保证 n ≤ 1000, hi ≤ 10^6。
对于 100% 的数据，保证 n ≤ 2 × 105, hi ≤ 10^18

代码实现： 

#include<iostream>
#include<cstring>
#include<cstdio>
#include<algorithm>
#include<cmath>

using namespace std;

const int N = 1e6 +10;
int n;
int ban[N];
int res;

int main(){
    cin >> n;
    for(int i = 0; i < n; i ++)
        scanf("%d", &ban[i]);
    while(true){
        int maxv = 0;
        for(int i = 0; i < n; i ++)
            if(maxv < ban[i]) maxv = ban[i];
        if(maxv == 1) break;
        res ++;
        bool flag = false;
        for(int i = 0; i < n; i ++){
            if(ban[i] == maxv){
                ban[i] = sqrt(ban[i] / 2 + 1);
                flag = true;
            }
            else if(flag && ban[i] != maxv) break;
        }
    }
    cout << res << endl;
}
